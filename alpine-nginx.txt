### Alpine Linux nginx configuration

apk add nginx

rc-update add nginx default

rc-service nginx start

vi /etc/nginx/nginx.conf

https://raw.githubusercontent.com/ecamposh/cloud-servers/main/nginx.conf

mkdir -p /var/www/html

echo "<h1>Hello, Nginx on Alpine!</h1>" > /var/www/html/index.html

chown -R nginx:nginx /var/www/html

chmod -R 755 /var/www/html

nginx -t

rc-service nginx reload

curl http://localhost

apk add iptables

iptables -A INPUT -p tcp --dport 80 -j ACCEPT
